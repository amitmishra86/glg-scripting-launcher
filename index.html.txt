<script>
$(function () {

  // Left panel links
  $("button[data-url]").on("click", function () {
    window.open($(this).data("url"), "_blank");
  });

  // Right panel actions
  $(".action").on("click", function () {
    let pid = $("#pid").val().trim();
    if (!pid) {
      alert("Please enter Project ID");
      return;
    }

    let type = $(this).data("type");
    let url = "";

    if (type === "quota") url = `https://glg.decipherinc.com/rep/selfserve/${pid}:dashboard?tab=quota`;
    if (type === "builder") url = `https://glg.decipherinc.com/apps/lumos/${pid}:edit`;
    if (type === "xml") url = `https://glg.decipherinc.com/apps/lumos/${pid}:xmledit`;
    if (type === "viewxml") url = `https://glg.decipherinc.com/apps/lumos/${pid}/temp-view:xmledit`;
    if (type === "lang") url = `https://glg.decipherinc.com/apps/mls/selfserve/${pid}:edit`;
    if (type === "test") url = `https://glg.decipherinc.com/survey/selfserve/${pid}`;
    if (type === "rdg") url = `https://glg.decipherinc.com/admin/sst/sst?path=selfserve/${pid}`;
    if (type === "download") url = `https://glg.decipherinc.com/report/selfserve/${pid}?run=1`;
    if (type === "edit") url = `https://glg.decipherinc.com/rep/selfserve/${pid}:edit`;
    if (type === "settings") url = `https://glg.decipherinc.com/apps/lumos/${pid}:edit#fn=openGlobalPreferences`;

    window.open(url, "_blank");
  });

});
</script>
